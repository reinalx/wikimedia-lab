package com.learn.wikimedialab.application.interceptor;

import com.learn.wikimedialab.domain.exceptions.base.BaseWikimediaLabException;
import com.learn.wikimedialab.domain.exceptions.base.NotFoundWikimediaLabException;
import java.util.HashMap;
import java.util.Map;
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.ExceptionHandler;
import org.springframework.web.bind.annotation.RestControllerAdvice;

/**
 * Global exception handler for handling Wikimedia Lab domain exceptions.
 */
@RestControllerAdvice
public class GlobalExceptionHandler {

  /**
   * Handles BaseWikimediaLabException and returns an appropriate error response.
   *
   * @param ex the BaseWikimediaLabException to handle
   * @return a ResponseEntity containing the error response map and HTTP status
   */
  @ExceptionHandler(BaseWikimediaLabException.class)
  public ResponseEntity<Map<String, String>> handleVisitsException(BaseWikimediaLabException ex) {
    return this.buildErrorResponse(ex, HttpStatus.INTERNAL_SERVER_ERROR);
  }

  /**
   * Handles NotFoundWikimediaLabException and returns an appropriate error response.
   *
   * @param ex the NotFoundWikimediaLabException to handle
   * @return a ResponseEntity containing the error response map and HTTP status
   */
  @ExceptionHandler(NotFoundWikimediaLabException.class)
  public ResponseEntity<Map<String, String>> handleVisitsException(
      NotFoundWikimediaLabException ex) {
    return this.buildErrorResponse(ex, HttpStatus.NOT_FOUND);
  }

  /**
   * Builds a ResponseEntity containing the error response map and the specified HTTP status.
   *
   * @param ex     the BaseWikimediaLabException containing error details
   * @param status the HTTP status to be set in the response
   * @return a ResponseEntity with the error response map and HTTP status
   */
  private ResponseEntity<Map<String, String>> buildErrorResponse(BaseWikimediaLabException ex,
      HttpStatus status) {
    final Map<String, String> errorResponse = this.generateErrorResponse(ex.getCode(),
        ex.getDescription());
    return ResponseEntity.status(status).body(errorResponse);
  }

  /**
   * Generates an error response map containing the provided error code and message.
   *
   * @param errorCode    the code representing the error
   * @param errorMessage the message describing the error
   * @return a map with keys "errorCode" and "errorMessage" containing the respective values
   */
  private Map<String, String> generateErrorResponse(String errorCode, String errorMessage) {
    final Map<String, String> errorResponse = new HashMap<>();
    errorResponse.put("errorCode", errorCode);
    errorResponse.put("errorMessage", errorMessage);
    return errorResponse;
  }
}
